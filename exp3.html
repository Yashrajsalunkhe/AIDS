Experiment No: 3

Title: JavaScript Operators

Aim:

Understanding JavaScript Operators like Arithmetic, Assignment, and Ternary Operators.

Examples:

1. Assignment Operator (=)

        <script>
        let x = 5;
        let y = 2;
        let z = x + y;
        document.getElementById("demo").innerHTML = "The sum of x + y is: " + z;
        </script>

2. Addition Operator (+)

        <script>
        let x = 5;
        let y = 2;
        let z = x + y;
        document.getElementById("demo").innerHTML = z;
        </script>

3. Multiplication Operator (*)

        <script>
        let x = 5;
        let y = 2;
        let z = x * y;
        document.getElementById("demo").innerHTML = z;
        </script>

4. JavaScript Functions for Arithmetic

        <script>
        function mul(p1, p2) { return p1 * p2; }
        function add(p1, p2) { return p1 + p2; }
        function sub(p1, p2) { return p1 - p2; }
        function div(p1, p2) { return p1 / p2; }
        let result = add(4, 3);
        document.getElementById("demo").innerHTML = result;
        </script>

5. Ternary Operator

        <script>
        function myFunction() {
            let age = document.getElementById("age").value;
            let voteable = (age < 18) ? "Too young" : "Old enough";
            document.getElementById("demo").innerHTML = voteable + " to vote.";
        }
        </script>
